#include<stdio.h>

            int main() {
                int n;
                
              
                printf("Enter the size of the array: ");
                scanf("%d", &n);
            
                int arr[n];
            
                
                printf("Enter %d integers:\n", n);
                for (int i = 0; i < n; i++) {
                    scanf("%d", &arr[i]);
                }
            
                
                for (int i = 0; i < n; i++) {
                    arr[i] = arr[i] * arr[i];
                }
            
                
                printf("Squared values of the array:\n");
                for (int i = 0; i < n; i++) {
                    printf("%d ", arr[i]);
                }
                
            
                return 0;
            }

            

                
-----OUTPUT-----

                Enter the size of the array: 3
                Enter 3 integers:
                2 4 5
                Squared values of the array:
                4 16 25
