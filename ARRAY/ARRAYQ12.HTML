#include<stdio.h>
            int main() {
                int n, i, j;
                printf("Enter the size of the array: ");
                scanf("%d", &n);
            
                int arr[n];
            
                printf("Enter %d elements:\n", n);
                for (i = 0; i < n; i++) {
                    scanf("%d", &arr[i]);
                }
            
                for (i = 0; i < n; i++) {
                    for (j = i + 1; j < n; j++) {
                        if (arr[i] == arr[j]) {
                            arr[j] = 0; 
                        }
                    }
                }
            
                printf("Array with duplicates removed and filled with zeros:\n");
                for (i = 0; i < n; i++) {
                    printf("%d ", arr[i]);
                }
            
                return 0;
            }
