#include<stdio.h>

            int main() {
                int n;
            
                printf("Enter the number of elements in the array: ");
                scanf("%d", &n);
            
                if (n <= 0) {
                    printf("Invalid input. Please enter a positive number of elements.\n");
                    return 1;
                }
            
                int arr[n];
            
                printf("Enter %d numbers:\n", n);
                for (int i = 0; i < n; i++) {
                    scanf("%d", &arr[i]);
                }
            
                printf("Positive numbers in the array are: ");
                for (int i = 0; i < n; i++) {
                    if (arr[i] > 0) {
                        printf("%d ", arr[i]);
                    }
                }
            
                printf("\n");
            
                return 0;
            }

                Enter 4 numbers:
                0
                1
