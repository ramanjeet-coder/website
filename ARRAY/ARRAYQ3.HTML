#include<stdio.h>
        int main() {
            int size;
            printf("Enter the size of the array: ");
            scanf("%d", &size);
            int firstArray[size];
            int secondArray[size];
            printf("Enter %d elements for the first array:\n", size);
            for (int i = 0; i < size; i++) {
                scanf("%d", &firstArray[i]);
            }
            for (int i = 0; i < size; i++) {
                secondArray[i] = firstArray[i];
            }
            printf("Second Array: ");
            for (int i = 0; i < size; i++) {
                printf("%d ", secondArray[i]);
            }
            printf("\n");
            return 0;
        }

            1
            2

            5
