#include<stdio.h>

    int main() {
        int n;
    
        printf("Enter the number of elements in the array: ");
        scanf("%d", &n);
    
        int arr[n];
    
        printf("Enter %d elements in the array:\n", n);
        for (int i = 0; i < n; i++) {
            scanf("%d", &arr[i]);
        }
    
        printf("Even numbers in the array: ");
        for (int i = 0; i < n; i++) {
            if (arr[i] % 2 == 0) {
                printf("%d ", arr[i]);
            }
        }
    
        return 0;
    }

-----OUTPUT-----

Enter the number of elements in the array: 6 Enter 6 elements in the array: 5 4 6 7 9 2 Even numbers in the array: 4 6 2
