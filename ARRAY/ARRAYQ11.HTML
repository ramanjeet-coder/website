#include 

            int main() {
                int n, i, j;
            
                printf("Enter the size of the array: ");
                scanf("%d", &n);
            
                int arr[n];
            
                printf("Enter %d elements:\n", n);
                for (i = 0; i < n; i++) {
                    scanf("%d", &arr[i]);
                }
            
                printf("Duplicate values in the array are: ");
                for (i = 0; i < n; i++) {
                    for (j = i + 1; j < n; j++) {
                        if (arr[i] == arr[j]) {
                            printf("%d ", arr[i]);
                            break; 
                        }
                    }
                }
            
                return 0;
            }
