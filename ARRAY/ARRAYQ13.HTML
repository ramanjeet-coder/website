#include<stdio.h>

            int main() {
                int n, i, j;
            
                printf("Enter the size of the array: ");
                scanf("%d", &n);
            
                int arr[n];
            
                printf("Enter %d elements:\n", n);
                for (i = 0; i < n; i++) {
                    scanf("%d", &arr[i]);
                }
            
                int count = 0;
                for (i = 0; i < n; i++) {
                    int isDuplicate = 0;  
            
                    for (j = i + 1; j < n; j++) {
                        if (arr[i] == arr[j]) {
                            isDuplicate = 1;
                            break;  
                        }
                    }
            
                    if (isDuplicate) {
                        count++;
                    }
                }
            
                printf("Number of duplicate values in the array: %d\n", count);
            
                return 0;
            }
