#include<stdio.h>>

            int main() {
                int n;
            
                printf("Enter the number of elements in the array: ");
                scanf("%d", &n);
            
                int arr[n];
            
                printf("Enter the array elements:\n");
                for (int i = 0; i < n; i++) {
                    scanf("%d", &arr[i]);
                }
            
                int sum = 0;
            
                for (int i = 0; i < n; i++) {
                    sum += arr[i];
                }
            
                printf("Sum of all values in the array is: %d\n", sum);
            
                return 0;
            }

            

            
-----OUTPUT-----

            Enter the number of elements in the array: 4
            Enter the array elements:
            1
            2
            3
            4
            Sum of all values in the array is: 10
